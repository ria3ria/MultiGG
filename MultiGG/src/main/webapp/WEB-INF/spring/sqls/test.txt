CREATE TABLE LOLBOARD(
	BOARDKATEGORIE VARCHAR(100) NOT NULL,
	BOARDNO INT AUTO_INCREMENT PRIMARY KEY,
	BOARDNAME VARCHAR(1000) NOT NULL,
	BOARDTITLE VARCHAR(2000) NOT NULL,
	BOARDCONTENT VARCHAR(2000) NOT NULL,
	BOARDDATE DATE NOT NULL
);


DROP TABLE MEMBER;

CREATE TABLE MEMBER(
	MEMBERNO INT AUTO_INCREMENT PRIMARY KEY,
	MEMBEREMAIL VARCHAR(1000) NOT NULL,
	MEMBERPW VARCHAR(1000) NOT NULL,
	MEMBERNAME VARCHAR(10000) NOT NULL,
	MEMBERLEVEL VARCHAR(10) NOT NULL,
	MEMBERDATE DATE NOT NULL
);
